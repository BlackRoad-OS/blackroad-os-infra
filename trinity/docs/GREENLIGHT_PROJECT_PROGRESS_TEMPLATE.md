```bash

    local project="$1"
    local phase="$2"       # epic, strategy, build, test, review, deploy, close
    local progress="$3"    # 0-7 (number of completed units)
    local status="$4"      # green, yellow, orange, red, none
    local sentiment="$5"   # Current mood emoji
    
    gl_log "ðŸ§­${phase}${progress}/7${status}" \
        "phase_progress" \
        "$project" \
        "Phase: ${phase}, Progress: ${progress}/7, Status: ${status}, Sentiment: ${sentiment}"

```

### NATS Subject Pattern
```
greenlight.project.{project_id}.phase.{phase_number}.{status}

Examples:
greenlight.project.api-v2.phase.3.yellow     # Build phase, yellow status
greenlight.project.ui-v2.phase.4.orange      # Test phase, orange status
greenlight.project.deploy.phase.6.red        # Deploy phase, red status
```

### Database Schema
```sql
CREATE TYPE project_phase AS ENUM (
  'epic', 'strategy', 'build', 'test', 'review', 'deploy', 'close'
);

CREATE TYPE phase_status AS ENUM (
  'green', 'yellow', 'orange', 'red', 'none'
);

CREATE TABLE project_progress (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  project_id UUID NOT NULL,
  phase project_phase NOT NULL,
  progress INTEGER NOT NULL CHECK (progress >= 0 AND progress <= 7),
  status phase_status NOT NULL DEFAULT 'none',
  sentiment TEXT,
  updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  notes TEXT
);
```
